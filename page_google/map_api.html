取得mapId

mapId不能用styles

      
        <style>
          #map {
            height: 400px;
            width: 100%;
          }
        </style>
        <div id="map"></div>
        <script>
          function initMap() {
            var uluru = {lat: 25.0253124, lng: 121.544109};
            var map = new google.maps.Map(document.getElementById('map'), {
              zoom: 16, 
              heading: 320,//斜度
              tilt: 47.5,//斜度
              center: uluru,
              mapId: YOUR_MAPId,
              mapTypeControl: false, //地圖UI隱藏
              streetViewControl: false,//小人UI隱藏
              fullscreenControl: false,//放大UI隱藏
            });
            
            //顯示地標ICON
            var marker = new google.maps.Marker({
              position: uluru,
              map: map
            });
          }
        </script>
        <script async defer src="https://maps.googleapis.com/maps/api/js?key=YOUR_API_KEY&callback=initMap"></script>
