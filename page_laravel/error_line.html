error_line

Line開網頁會不執行vite app.js 解法就是在body底下重新加載

      
        <script>
          //判斷裝置是Line 就重新加載
          const isLine = /Line/i.test(navigator.userAgent);
          if (isLine && !sessionStorage.getItem('lineFirstLoad')) {
            // console.log('isLine', isLine)
            // alert('isLine',isLine);
            sessionStorage.setItem('lineFirstLoad', 'true');
            window.location.reload();
          }
        </script>
        @vite(['resources/js/app.js'])
