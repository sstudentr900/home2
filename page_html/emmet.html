前端開發神器 Emmet

div#container>p.title+div.body>span.snippet+img[src="wfublog.jpg"]

    
      <!-- 產生HTML -->
      <div id="container">
        <p class="title"></p>
        <div class="body">
          <span class="snippet"></span>
          <img src="wfublog.jpg" alt="">
        </div>
      </div>

括號 () 搭配乘號 *

#WFU>ul>(li>a)*3

    
      <!-- 產生HTML -->
      <div id="WFU">
        <ul>
          <li><a href=""></a></li>
          <li><a href=""></a></li>
          <li><a href=""></a></li>
        </ul>
      </div>

^ (回到上一層)

div>p^div

    
      <!-- 產生HTML -->
      <div>
        <p></p>
      </div>
      <div></div>

續編功能 $

ul>.wfu$*3

    
      <!-- 產生HTML -->
      <ul>
        <li class="wfu1"></li>
        <li class="wfu2"></li>
        <li class="wfu3"></li>
      </ul>

ul>.wfu$$*3

    
      <!-- 產生HTML -->
      <ul>
        <li class="wfu01"></li>
        <li class="wfu02"></li>
        <li class="wfu03"></li>
      </ul>

續編功能指定起始數從 5 開始編 @

ul>li.wfu$@5*3

    
      <!-- 產生HTML -->
      <ul>
        <li class="wfu5"></li>
        <li class="wfu6"></li>
        <li class="wfu7"></li>
      </ul>

[](設定屬性)

div[titl border='1']

    
      <!-- 產生HTML -->
      <div titl="" border="1"></div>

{}(輸入內容)　　注意：此div不可省略!!!

div{hello}

    
      <!-- 產生HTML -->
      <div>hello</div>
