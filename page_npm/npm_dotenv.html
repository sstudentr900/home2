npm dotenv 環境變數

    
      //terminal
      npm i dotenv

      //新增.env檔，建立環境變數
      YOUR_VARIABLE='這是環境變數'

      //在js引入環境變數
      require('dotenv').config(); //載入.env環境檔
      function getEnvVariable () {
          const env_variable= process.env.YOUR_VARIABLE // 取出環境變數
          console.log(env_variable)
      }
      getEnvVariable()
