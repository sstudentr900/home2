      setup() {
        const msg = ref('Hello World');
        return { msg }
      }

        const msg = ref('Hello World');
      </script>

ref()

  • ref 包裝該值讓他有響應試
  • ref 是obj請用const
  • ref 被包裝後，要用.value才能取值
  • ref 可以包任何型別
    
      <template>
        <h1>vue: {{ count }}</h1>
        <button @click="increment"> count is: {{ count }}</button>
      </template>
      <script>
        const count = ref(0); 
        const increment = ()=>{
          //count 被包裝後需要用.value才能取值
          console.log(count)
          count.value += 1;
        }
      </script>

reactive()

  • reactive 可以包{}和[]
    
      <template>
        <h1>vue: {{ data }}</h1>
      </template>
      <script>
        const data = reactive({
          name: 'mike'
        }); 
        setTimeout(()=>{
          data.name = 'jacky'
        },2000)
      </script>
