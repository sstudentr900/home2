ECharts关于y轴刻度调整

      
        1.
        min:0,  //取0为最小刻度
        max: 100, //取100为最大刻度

        2.
        min:'dataMin', //取最小值为最小刻度
        max: 'dataMax', //取最大值为最大刻度

        3.
        min: function(value) {//取最小值向下取整为最小刻度
            return Math.floor(value.min)
        },
        max: function(value) {//取最大值向上取整为最大刻度
            return  Math.ceil(value.max) 
        },

        4.
        scale: true, //自适应
        minInterval: 0.1, //分割刻度
