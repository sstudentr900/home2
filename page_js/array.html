          // 把二维数组扁平化---------------------------------------------------------------
          var myArray = [
              [0, 8],

              [4, 9]

          var myNewArray = [];
          for (var i = 0; i < myArray.length; ++i) {

          //Array创建新方法(toUpperCase)-------------------------------------------------------
          Array.prototype.myUcase = function() {
            for (i = 0; i < this.length; i++) {
              this[i] = this[i].toUpperCase();
            }
          }

          function myFunction() {
              var fruits = ["Banana", "Orange", "Apple", "Mango"];
              fruits.myUcase();
              console.log(fruits);
          }

          //array-雙指針---------------------------------------------------------
          var arr = [3, 3, 3, 4, 2, 2, 2, 4];
          var i = 0;
          var j = 1;
          var result = [];
          while (i < 8) {
            if (arr[i] == arr[j]) {
              j++;
            } else {
              if (j - i >= 3) {
                  for (var m = i; m < j; m++) {
                      result.push(m);
                  }
              }
              i = j;
              j++;
            }
          }
