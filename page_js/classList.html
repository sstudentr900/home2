indexOf

        
            function hasClass(obj, cls) {
                return (obj.className).indexOf(cls) > -1
            }

            function main() {
                var obj = document.getElementById('box');
                if (hasClass(obj, 'active')) {
                    obj.className = '';
                } else {
                    obj.className = 'active'
                }
            }

contains

        
            function hasClass() {
                var obj = document.querySelector('#box');
                if (obj.classList.contains('active')) {
                    obj.className = '';
                } else {
                    obj.className = 'active'
                }
            }

            function hasClass() {
                var obj = document.querySelector('#box');
                if (obj.matches('.active')) {
                    obj.className = '';
                } else {
                    obj.className = 'active'
                }
            }
