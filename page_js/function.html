立刻執行的函式寫法就稱為IIFE

      
        <script>
            var hello2 = function(name) {
              console.log('hello2')
              return 'Hello ' + name
            }('Simon');
            console.log(hello2);

            //匿名自我调用的函数
            (function(i) {
                var x = "Hello!!"; // 我将调用自己
            })(i);

            //和上面一樣
            !function(i) {
                var x = "Hello!!"; // 我将调用自己
            }(i);
        </script>

函數傳遞參數

      
        <script>
        //函數傳遞參數(用object較好)
        function person(obj) {
            console.log(obj.name);
            console.log(obj.age);
        }
        var data = {
            name: 'andy',
            age: '27'
        };
        person(data);
        //'andy'
        //'27'

        //es5
        function Person(name, age) {
            this.name = name || 'person';
            this.age = age || 18;
        }
        //es6
        function Person(name = 'person', age = 18) {
            this.name = name;
            this.age = age;
        }

        //ES6 其餘參數 直接傳入剩餘未定的變數內容
        //https://ithelp.ithome.com.tw/articles/10241916
        const callMore = (a, ...args) => {
            console.log(args);//['b','c']
        }
        callMore('a', 'b', 'c');
        </script>

arguments轉成數組apply

      
        <script>
        //apply
        function show(){
          var args=Array.prototype.concat.apply([1,2,3], arguments);
          console.log(args);
        }
        show(1,2,3,4);//[1, 2, 3, 1, 2, 3, 4]

        //apply
        function show(){
          var args=Array.prototype.slice.apply(arguments);
          console.log(args);
        }
        show(1,2,3,4);//[1, 2, 3, 4]

        //for
        var args = [];
        for (var i = 0; i < arguments.length; i++) {
          args.push(arguments[i]);
        }

        //[...arguments]
        //https://www.casper.tw/javascript/2017/12/26/javascript-array/
        const callAll = function() {
          // console.log(arguments);
          let thisPeople = [...arguments];
        }
        callAll(1, 2, 3);


        //Array.from(arguments)
        function sumArguments() {
              return Array.from(arguments).reduce((sum, num) => sum + num);
          }
          sumArguments(1, 2, 3); // => 6
        </script>

        //3.

            greet: 'Good morning'
        }); //Good morning, undefined
        </script>

        }
        f() // undefined
        //3.
