es6 spread operator 扩展运算符

字符串转为数组

      
        <script>
          [...'hello']// [ "h", "e", "l", "l", "o" ]
        </script>

合并array

      
        <script>
          var arr1 = ['a', 'b'];
          var arr2 = ['c'];
          var arr3 = ['d', 'e'];
          var arr4 =[...arr1, ...arr2, ...arr3]; //['a','b','c','d', 'e']
        </script>

複製陣列不變原陣列

      
        <script>
          let today = ['沙拉', '水煮雞胸'];
          let tomorrow = [...today];
          console.log(tomorrow); //['沙拉', '水煮雞胸']
          console.log(today===tomorrow); //false
        </script>

把陣列傳入函式做運算

      
        <script>
          function count(breakfast, lunch, dinner) {
              return breakfast + lunch + dinner;
          }

          var price = [10, 50, 20];
          console.log(count(...price)); // 80
        </script>

把二维数组扁平化

      
        <script>
          let numbers2 = [1, 2, 3];
          let newNumbers = [...numbers2, 4]; //新的array
          console.log(newNumbers);//[1, 2, 3, 4]
        </script>

数据结构转为数组

      
        <script>
          [...document.querySelectorAll('div')]
        </script>

遍历器（Iterator）接口的对象，都可以用扩展运算符转为真正的数组

      
        <script>
          let nodeList = document.querySelectorAll('div');
          let array = [...nodeList];
        </script>

new Map

      
        <script>
          let map = new Map([
            [1, 'one'],
            [2, 'two'],
            [3, 'three'],
          ]);

          let arr = [...map.keys()]; // [1, 2, 3]
        </script>
