点击网页脚本教学

    
        // 获取需要点击的按钮元素
        const button = document.querySelector('#button');
        // 模拟鼠标点击事件
        const event = new MouseEvent("click", {
          view: window,
          bubbles: true, //指定事件是否应该冒泡。这里设为 true。
          cancelable: true //指定事件是否可以被取消。这里设为 true。
        });
        button.dispatchEvent(event);
