object

      
        <style>
          .fit_contain{
            /* 圖片大小:維持比例*/
            object-fit: contain;
          }
          .fit_cover{
            /* 圖片大小:填滿*/
            object-fit: cover;
          }
          .position_center{
            /* 圖片位置:置中 */
            object-position: center;
          }
          .position_top{
            /* 圖片位置:置上 */
            object-position: top;
          }
        </style>
