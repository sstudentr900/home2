                //單行
                .ellipsis {
                    overflow:hidden;
                    white-space: nowrap;
                    text-overflow: ellipsis;
                }

                //單行
                .ellipsis1 {
                    overflow:hidden;
                    white-space: normal;
                    display: -webkit-box;
                    /* 行數 */
                    -webkit-line-clamp: 1;
                    -webkit-box-orient: vertical;
                }

                //多行
                .ellipsis2{
                    overflow:hidden;
                    white-space: normal;
                    display: -webkit-box;
                    /* 行數 */
                    -webkit-line-clamp: 2;
                    -webkit-box-orient: vertical;
                }
